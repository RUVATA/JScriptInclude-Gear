EXAMPLE
//  [ENG]                
//  Above this code should be defined JScriptInclude Gear      
//  The "email" module must be defined in config file of JScriptInclude Gear
//  [RUS]                
// Перед этим кодом должен быть обявлен механизм JScriptInclude Gear  
// модуль "email"  должен быть определен в файле конфигурации JScriptInclude Gear
                   
jsImport("email");              
eval(jsImport.initialization);            
                   
var mailer = new Mail.GMail({            
                sendusername: "someuser@gmail.com",      
                sendpassword: "somepassword"        
        });                
                   
mailer.addAttachment("D:\\SomeDir\\TestAttachment.file");     
mailer.send({                
        from: "SomeUser of GMail",          
        to: "targetuser@gmail.com",          
        subject: "Testing of JScriptInclude email module",    
        textbody: "Hello World!"          
});
